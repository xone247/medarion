# Kill Process on Port 44047 - Run in Jupyter Terminal

# Option 1: Find and kill process on port 44047
lsof -ti :44047 | xargs kill -9
# OR
fuser -k 44047/tcp

# Option 2: Find what's using port 44047
lsof -i :44047
# OR
netstat -tulpn | grep 44047

# Verify port is free
lsof -i :44047

# If nothing shows, port is free!
