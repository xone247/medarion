# Kill Process on Port 8080 - Run in Jupyter Terminal

# Option 1: Find and kill process on port 8080
lsof -ti :8080 | xargs kill -9
# OR
fuser -k 8080/tcp

# Option 2: Find what's using port 8080
lsof -i :8080
# OR
netstat -tulpn | grep 8080

# Option 3: Kill all Python processes (use with caution)
# pkill -9 python3

# Verify port is free
lsof -i :8080

# If nothing shows, port is free!
